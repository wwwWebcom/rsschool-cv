                        "const progressId = document.querySelector("#progress");
                        const btnPrevId = document.querySelector("#prev");
                        const btnNextId = document.querySelector("#next");
                        const circles = document.querySelectorAll(".circle");
                        let width = 0;
                        let currentActive = 1;
                        btnNextId.addEventListener("click", () => {
                          currentActive++;
                          update();
                          progressId.style.width = `${(width += 33)}%`;
                          count();
                        });
                        btnPrevId.addEventListener("click", () => {
                          currentActive--;
                          update();
                          progressId.style.width = `${(width -= 33)}%`;
                          count();
                        });
                        
                        function count() {
                          if (width === 99) {
                            btnNextId.disabled = true;
                          } else {
                            btnNextId.disabled = false;
                          }
                          if (width === 0) {
                            btnPrevId.disabled = true;
                          } else {
                            btnPrevId.disabled = false;
                          }
                        }
                        
                        function update() {
                          circles.forEach((circle, idx) => {
                            if (idx < currentActive) {
                              circle.classList.add("active");
                            } else {
                              circle.classList.remove("active");
                            }
                          });
                        }"
